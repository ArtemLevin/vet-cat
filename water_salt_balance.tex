\section{Введение в тему}
Это пособие объединяет три важных блока подготовки к ЕГЭ:
\begin{itemize}
    \item качественные реакции в органической химии;
    \item признаки реакций ионного обмена в неорганической химии;
    \item расчёты по уравнениям реакций (растворы, тепловой эффект, выход продукта).
\end{itemize}
Материал построен так, чтобы можно было повторять теорию и сразу применять её в типовых заданиях.

\section{Основные понятия и термины}
\begin{itemize}
    \item \textbf{Качественная реакция} --- реакция, позволяющая распознать класс вещества по характерному признаку (осадок, газ, изменение цвета, «серебряное зеркало» и т.д.).
    \item \textbf{Признак реакции} --- наблюдаемое изменение: выделение газа, образование осадка, изменение окраски, выделение металла.
    \item \textbf{Теоретический выход} --- количество (масса) продукта, рассчитанное строго по уравнению реакции.
    \item \textbf{Практический выход} --- реально полученное количество (масса) продукта.
    \item \textbf{Выход продукта}:
    \[
    \eta = \frac{m_{\text{практ}}}{m_{\text{теор}}}\cdot 100\%.
    \]
\end{itemize}

\subsection{Ключевые уравнения}
\begin{itemize}
    \item Карбоновая кислота и гидрокарбонат:
    \[
    \mathrm{CH_3COOH + NaHCO_3 \rightarrow CH_3COONa + H_2O + CO_2\uparrow}.
    \]
    \item Этанол и оксид меди(II):
    \[
    \mathrm{CH_3CH_2OH + CuO \xrightarrow{t} CH_3CHO + Cu + H_2O}.
    \]
    \item Альдегид и аммиачный раствор оксида серебра (схематично):
    \[
    \mathrm{R{-}CHO \rightarrow R{-}COOH}, \quad \mathrm{Ag^+ \rightarrow Ag\downarrow}.
    \]
\end{itemize}

\subsection{Важная оговорка}
Муравьиная кислота \(\mathrm{HCOOH}\) является исключением: она может давать реакцию «серебряного зеркала». Для большинства карбоновых кислот эта реакция нехарактерна.

\section{Качественные реакции ионного обмена в неорганике}

\subsection{Когда реакция ионного обмена протекает}
Реакция в водном растворе протекает до конца, если образуется хотя бы один из продуктов:
\begin{itemize}
    \item осадок;
    \item газ;
    \item слабый электролит (например, вода).
\end{itemize}

\subsection{Типовые примеры}
\begin{itemize}
    \item \(\mathrm{CuSO_4 + 2NaOH \rightarrow Cu(OH)_2\downarrow + Na_2SO_4}\): голубой осадок.
    \item \(\mathrm{CuSO_4 + Na_2CO_3 + H_2O \rightarrow Cu(OH)_2\downarrow + CO_2\uparrow + Na_2SO_4}\): осадок и газ (с учётом гидролиза карбоната меди).
    \item \(\mathrm{CuSO_4 + H_2S \rightarrow CuS\downarrow + H_2SO_4}\): чёрный осадок сульфида меди(II), исчезновение голубой окраски раствора.
\end{itemize}

\subsection{Случай, когда реакции нет}
\(\mathrm{CuSO_4}\) и разбавленная \(\mathrm{HNO_3}\) в школьной постановке задачи не реагируют: все возможные продукты растворимы, нет газа, осадка и слабого электролита.

\section{Схемы превращений (типовые цепочки)}
\begin{enumerate}
    \item \(\mathrm{CH_3CH_2OH \xrightarrow{CuO,\ t} CH_3CHO \xrightarrow{[O]} CH_3COOH \xrightarrow{NaHCO_3} CH_3COONa}\).
    \item \(\mathrm{C_2H_4 \xrightarrow{Br_2} C_2H_4Br_2}\), далее гидролизом можно получить многоатомный спирт (школьный обзорный уровень).
    \item \(\mathrm{Fe \xrightarrow{HNO_3(конц.),\ t} Fe(NO_3)_3}\), побочный газ --- \(\mathrm{NO_2}\).
\end{enumerate}

\section{Алгоритмы решения расчётных задач}

\subsection{Задачи на растворы}
\textbf{Шаги:}
\begin{enumerate}
    \item Выписать, что дано: масса раствора, массовая доля, масса растворённого вещества.
    \item Использовать формулу \(\omega = \frac{m_{\text{в-ва}}}{m_{\text{р-ра}}}\).
    \item Проверить, не перепутаны ли масса раствора и масса вещества.
    \item Проверить ответ на здравый смысл: \(0<\omega<1\) (или от 0\% до 100\%).
\end{enumerate}

\subsection{Задачи на тепловой эффект реакции}
\textbf{Шаги:}
\begin{enumerate}
    \item Составить уравнение реакции.
    \item Определить теплоту, соответствующую «теоретическому» количеству вещества по уравнению.
    \item Через пропорцию связать реальную теплоту по условию и искомое количество вещества.
    \item Перевести количество вещества газа в объём (при н.у. \(22{,}4\ \mathrm{л/моль}\)).
\end{enumerate}

\subsection{Задачи на выход продукта}
\textbf{Шаги:}
\begin{enumerate}
    \item Найти теоретическое количество (массу) продукта по уравнению.
    \item Сравнить с практической массой.
    \item Вычислить выход по формуле \(\eta = \frac{m_{\text{практ}}}{m_{\text{теор}}}\cdot 100\%\).
\end{enumerate}

\section{Типовые примеры}

\subsection*{Пример 1. Масса железа по объёму \(\mathrm{NO_2}\)}
Уравнение:
\[
\mathrm{Fe + 6HNO_3 \rightarrow Fe(NO_3)_3 + 3NO_2 + 3H_2O}.
\]
Если \(V(\mathrm{NO_2})=16{,}8\ \mathrm{л}\) (н.у.), то
\[
 n(\mathrm{NO_2})=\frac{16{,}8}{22{,}4}=0{,}75\ \mathrm{моль}.
\]
По уравнению \(1\) моль \(\mathrm{Fe}\) даёт \(3\) моль \(\mathrm{NO_2}\), значит
\[
 n(\mathrm{Fe})=\frac{0{,}75}{3}=0{,}25\ \mathrm{моль},
\]
\[
 m(\mathrm{Fe})=0{,}25\cdot 56=14\ \mathrm{г}.
\]

\subsection*{Пример 2. Выход метанола}
\[
\mathrm{CO + 2H_2 \rightarrow CH_3OH}.
\]
Дано: \(1\ \mathrm{м^3}\) \(\mathrm{CO}\) и \(2\ \mathrm{м^3}\) \(\mathrm{H_2}\), получено \(1\ \mathrm{кг}\) метанола.
Соотношение объёмов стехиометрическое \(1:2\).
\[
1000\ \mathrm{л\ CO} \Rightarrow n(\mathrm{CO})=\frac{1000}{22{,}4}=44{,}64\ \mathrm{моль}.
\]
Тогда
\[
 n_{\text{теор}}(\mathrm{CH_3OH})=44{,}64\ \mathrm{моль}, \quad
 m_{\text{теор}}=44{,}64\cdot 32\approx 1428{,}6\ \mathrm{г}=1{,}4286\ \mathrm{кг}.
\]
\[
 \eta=\frac{1{,}0}{1{,}4286}\cdot 100\%\approx 70\%.
\]

\section{Типичные ошибки}
\begin{itemize}
    \item Путаница реактива Толленса с реактивом на карбоновые кислоты.
    \item Игнорирование исключения: \(\mathrm{HCOOH}\) может вести себя как альдегид.
    \item Запись «реакция идёт», когда в растворе не образуются осадок, газ или слабый электролит.
    \item Неправильное уравнивание ОВР с \(\mathrm{HNO_3}\): пропуск коэффициента 6 перед кислотой.
    \item В расчётах: раннее округление, потеря размерности, подмена теоретической массы практической.
    \item Неверный выбор молярного объёма: \(22{,}4\ \mathrm{л/моль}\) применяют только для н.у.
\end{itemize}

\section{Что нужно помнить}
\begin{enumerate}
    \item Для карбоновых кислот надёжный школьный реактив --- \(\mathrm{NaHCO_3}\), признак --- \(\mathrm{CO_2}\uparrow\).
    \item Для альдегидов --- аммиачный раствор оксида серебра («серебряное зеркало»).
    \item Для многоатомных спиртов --- \(\mathrm{Cu(OH)_2}\) и ярко-синее окрашивание.
    \item В ионном обмене ищем \textit{осадок, газ, слабый электролит}.
    \item В расчётных задачах всегда фиксируйте: что дано, что найти, формула, размерность, проверка результата.
    \item Выход продукта: \(\eta = \frac{m_{\text{практ}}}{m_{\text{теор}}}\cdot 100\%\).
\end{enumerate}

\section*{Ответы}
\addcontentsline{toc}{section}{Ответы}

\begin{enumerate}
    \item \(\mathrm{NaHCO_3}\).

    \item \(\mathrm{CH_3COOH + NaHCO_3 \rightarrow CH_3COONa + H_2O + CO_2\uparrow}\);
    признак: выделение газа (пузырьки \(\mathrm{CO_2}\)).

    \item а) этаналь --- аммиачный раствор оксида серебра;
    б) глицерин --- \(\mathrm{Cu(OH)_2}\);
    в) этилен --- бромная вода;
    г) фенол --- бромная вода (обесцвечивание и белый осадок).

    \item \(\mathrm{CH_3CH_2OH + CuO \xrightarrow{t} CH_3CHO + Cu + H_2O}\).

    \item Признаки: осадок и выделение газа.
    \[
    \mathrm{CuSO_4 + Na_2CO_3 + H_2O \rightarrow Cu(OH)_2\downarrow + CO_2\uparrow + Na_2SO_4}.
    \]

    \item В школьной постановке: реакции нет (нет осадка, газа, слабого электролита).

    \item \(\mathrm{Fe + 6HNO_3 \rightarrow Fe(NO_3)_3 + 3NO_2 + 3H_2O}\).

    \item \(14\ \mathrm{г}\).

    \item \(\eta \approx 70\%\).

    \item \(n(\mathrm{SO_2})=\frac{4{,}2}{10{,}5}=0{,}4\ \mathrm{моль}\),
    \(V(\mathrm{SO_2})=0{,}4\cdot 22{,}4=8{,}96\ \mathrm{л}\).
\end{enumerate}
